smuggler = {
    "spawn": range(0, grid.height / 4),  // since the start of the grid is in the bottom left corner,
                                            we can use the height of the grid to determine the spawn
                                            range
    "speed": 1,
    "drugs": 1,
    "direction": "up",
    "speed increased": 2,
    "range of vision": max(grid.(height+width) // 10, 3)
    "when caught": speed = 0
    "when no drugs": dies
    "functions": {
        "flee_from_police": move away from police,
        "border_crossed": disappear, if border crossed,
        "move": check for police in range, if police in range, flee from nearest officer, else go to border,
                                           if border reached, disappear,
        "step": if drugs == 0, die, else move,
    }
}

police = {
    "spawn": range(3 * grid.height / 4, grid.height),
    "speed": 1,
    "drugs": 0,
    "direction": "random",
    "speed_increased": 3,
    "range of vision": max(grid.(height+width) // 8, 5)
    "when captured smuggler": drugs += 1
    "functions": {
        "move": check for smuggler in range, if smuggler in range, move towards nearest smuggler, else move randomly,
        "chase smuggler": move towards smuggler,
        "step": move,
    }
}
